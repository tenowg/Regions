18  public class RegionData extends YamlConfiguration {

28  	}

30  	/**

32  	 *
33  	 * @param volume of the configuration
34  	 * @return the Volume configuration node
35  	 */
36  	public final VolumeData get(String volume) {
37  			if (VOLUME == null) {
38  				return new VolumeData(this, volume);
39  			}
40  			return VOLUME;
41  	}
42  	
43  	@Override
44  	public void load() {
45  		try {
46  			super.load();
47  			VOLUME.load();
48  		} catch (ConfigurationException e) {
49  			Regions.getInstance().getLogger().log(Level.WARNING, "Error loading Feature configuration: ", e);
50  		}
51  	}

61  		}
62  	}
