137 		if (detectors.containsKey(clazz)) {
138 			return (T) detectors.get(clazz);
139 		}
140 
141 		Detector detector = null;
142 		try {
143 			detector = clazz.newInstance();
144 			detector.setFeature(this);
145 		} catch (InstantiationException | IllegalAccessException ex) {
146 			Spout.getLogger().log(Level.SEVERE, "Error create Detector Class " + clazz.getName());
147 		}
148 
149 		detectors.put(clazz, detector);
150 		return (T) detector;
151 	}
