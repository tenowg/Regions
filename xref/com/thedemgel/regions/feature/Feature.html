149 	public <T extends Detector> T add(Class<T> clazz) {
150 		if (detectors.containsKey(clazz)) {
151 			return (T) detectors.get(clazz);
152 		}
153 		
154 		Detector detector = null;
155 		try {
156 			detector = clazz.newInstance();
157 			detector.setFeature(this);
158 		} catch (InstantiationException | IllegalAccessException ex) {
159 			Spout.getLogger().log(Level.SEVERE, "Error create Detector Class " + clazz.getName());
160 		}
161 		
162 		detectors.put(clazz, detector);
163 		return (T) detector;
164 	}
