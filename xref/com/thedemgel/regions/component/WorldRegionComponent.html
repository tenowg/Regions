39  	/**
40  	 * Will add an already created Region to the Maps and parse its
41  	 * PointMaps and effectively add the region to the Map. (Should only be
42  	 * called from createRegion, or if the Region Object has been FULLY
43  	 * created first)
44  	 *
45  	 * @param region
46  	 */
47  	public void addRegion(Region region) {
48  		if (!regions.containsValue(region)) {
49  			getData().get("regions", new ConcurrentSkipListMap<UUID, Region>()).put(region.getUUID(), region);
50  			regions = getData().get("regions", new ConcurrentSkipListMap<UUID, Region>());
51  		}
52  
53  		updateRegion(region);
54  	}

57  	 * Will completely remove a region from all Maps effectively removing it
58  	 * from the game.
59  	 *
60  	 * @param region Region to be removed
61  	 */
62  	public void removeRegion(Region region) {
63  		for (PointMap mpoint : region.getPointCache()) {
64  			Set<Region> regs = xregions.get(mpoint);
65  			regs.remove(region);
66  		}
67  
68  		getData().get("regions", new ConcurrentSkipListMap<UUID, Region>()).remove(region.getUUID());
69  		regions = getData().get("regions", new ConcurrentSkipListMap<UUID, Region>());
70  	}

72  	/**
73  	 * Should be called after the size of a Region has been changed. This
74  	 * will "fix" the PointMaps and regenerate the PointMap cache of the
75  	 * region.
76  	 *
77  	 * @param region Region after it has been updated with new Coords.
78  	 */
79  	public Region updateRegion(Region region) {
80  
81  		if (region.getUUID() == null) {
82  			return null;
83  		}
84  
85  		for (PointMap mpoint : region.getPointCache()) {
86  			Set<Region> regs = xregions.get(mpoint);
87  			if (regs != null) {
88  				regs.remove(region);
89  			}
90  		}
91  
92  		region.resetPointCache();
93  
94  		int i, ii, iii;
95  
96  		for (i = (int) (region.getVolume().getLowX() / WorldRegionComponent.WORLD_GRID_SIZE); i <= (int) (region.getVolume().getHighX() / WorldRegionComponent.WORLD_GRID_SIZE); i++) {
97  			for (ii = (int) (region.getVolume().getLowY() / WorldRegionComponent.WORLD_GRID_SIZE); ii <= (int) (region.getVolume().getHighY() / WorldRegionComponent.WORLD_GRID_SIZE); ii++) {
98  				for (iii = (int) (region.getVolume().getLowZ() / WorldRegionComponent.WORLD_GRID_SIZE); iii <= (int) (region.getVolume().getHighZ() / WorldRegionComponent.WORLD_GRID_SIZE); iii++) {
99  					PointMap mpoint = new PointMap(i, ii, iii);
100 					Set<Region> regs = xregions.get(mpoint);
101 					if (regs == null) {
102 						regs = Collections.newSetFromMap(new ConcurrentHashMap<Region, Boolean>());
103 					}
104 					region.addPointCache(mpoint);
105 					regs.add(region);
106 					xregions.put(mpoint, regs);
107 				}
108 			}
109 		}
110 
111 		return region;
112 	}
